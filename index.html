body{ font-family: Arial, sans-serif; margin: 30px; } input, textarea, select{ width:100%; margin:10px 0; padding:10px; } button{ padding:10px 20px; cursor:pointer; } .card{ border:1px solid #ccc; padding:15px; margin-top:15px; } let posts = []; function addPost() { const title = document.getElementById("title").value; const category = document.getElementById("category").value; const content = document.getElementById("content").value; if(title === "" || content === ""){ alert("내용을 입력하세요."); return; } posts.push({ title, category, content, agree:0, disagree:0 }); displayPosts(); document.getElementById("title").value=""; document.getElementById("content").value=""; } function displayPosts(){ const postList = document.getElementById("postList"); postList.innerHTML=""; posts.forEach((post,index)=>{ postList.innerHTML += `

${post.title}

분야: ${post.category}

${post.content}

`; }); } function agree(index){ posts[index].agree++; displayPosts(); } function disagree(index){ posts[index].disagree++; displayPosts(); }
